SET(OUT_NAME SGMApp)
AUX_SOURCE_DIRECTORY(. SOURCE_LIST)
ADD_DEFINITIONS(${NORMAL_DEFINITION} -DNOMINMAX)

find_package(OpenCV REQUIRED)

INCLUDE_DIRECTORIES(
 ${OpenCV_INCLUDE_DIRS}
 ${CVL_INCLUDE_DIR}
)

LINK_DIRECTORIES(
 ${CVL_LIB_OUT_DIR}
)

ADD_EXECUTABLE(${OUT_NAME} ${SOURCE_LIST})

TARGET_LINK_LIBRARIES(${OUT_NAME} 
 ${OpenCV_LIBS}
)

SET_TARGET_PROPERTIES(${OUT_NAME} PROPERTIES FOLDER	Application)
set_target_properties(${OUT_NAME} PROPERTIES DEBUG_POSTFIX "d")

INSTALL(TARGETS ${OUT_NAME} RUNTIME DESTINATION bin)
